// --- CONFIGURATION ---
const FORM_LINK = "https://forms.gle/6uQqe4GHzujqCTqG8"; 
const DATA_TAB_NAME = "Student_Data";
const LOG_TAB_NAME = "Problem_Logs";

// --- 1. SEND FULL REPORTS TO ALL STUDENTS ---
function sendStudentReports() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(DATA_TAB_NAME);
  if (!sheet) return;
  
  const data = sheet.getDataRange().getValues();

  // Starting loop at row 2 (Index 2)
  for (let i = 2; i < data.length; i++) {
    const row = data[i];
    if (!row[17]) continue; // Skip if email is missing

    // --- DATA MAPPING & PERCENTAGE FIX ---
    const roll = row[0];        
    const name = row[1];        
    
    // Fix: Multiply by 100 if the value is a decimal (e.g., 0.85 -> 85)
    const fixPct = (val) => (val <= 1 && val > 0) ? Math.round(val * 100) : Math.round(val);

    const attVal = fixPct(row[6]);      
    const sem1 = fixPct(row[7]);        
    const sem2 = fixPct(row[8]);        
    const sem3 = fixPct(row[9]);        
    const avgVal = fixPct(row[10]);     
    
    const feeDue = row[15] || "Nil"; 
    const dashLink = row[16];        
    const email = row[17];           

    // --- AI Logic: Performance & Risk ---
    let performance = avgVal >= 75 ? "Excellent 🌟" : (avgVal >= 50 ? "Good 👍" : "Needs Improvement ⚠️");
    let riskPredictor = attVal < 75 ? "High Risk (Attendance below 75%) 🚩" : "Safe ✅";
    
    const subject = `Academic Performance & Progress Report - ${name}`;
    const message = `Dear ${name},

I hope this email finds you well. 

Please review your official academic and administrative summary below:

📌 STUDENT PROFILE:
- Roll Number: ${roll}
- Full Dashboard: ${dashLink}

📊 ACADEMIC PERFORMANCE:
- Semester 1: ${sem1}%
- Semester 2: ${sem2}%
- Semester 3: ${sem3}%
- Overall Average: ${avgVal}%
- Current Standing: ${performance}

🕒 ATTENDANCE & ELIGIBILITY:
- Current Attendance: ${attVal}%
- Risk Status: ${riskPredictor}

💰 ADMINISTRATIVE STATUS:
- Pending Fees: ${feeDue}

------------------------------------------------
SUPPORT & FEEDBACK:
Our goal is to ensure your success. If you are facing any challenges, technical errors, or personal issues in college, please use the link below to reach out to us:

Submit Feedback/Report Issue: ${FORM_LINK}

Upon submission, you will receive a unique Ticket ID to track the resolution status of your request.

Warm regards,
Academic Support Team
College Administration Office`;

    try {
      MailApp.sendEmail(email, subject, message);
      console.log(`Successfully sent to ${name}`);
    } catch (e) {
      console.log(`Error sending to ${name}: ${e.message}`);
    }
  }
}

// --- 2. AUTOMATIC TICKET GENERATION ---
function onFormSubmit(e) {
  if (!e) return;
  const responses = e.values; 
  const hasProblem = responses[1]; 
  const rollNo = responses[3];
  
  if (hasProblem.toLowerCase() !== "yes") return;

  const uniqueID = "TKT-" + Math.floor(1000 + Math.random() * 9000);
  const logSheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(LOG_TAB_NAME);
  const studentEmail = lookupEmail(rollNo);
  
  if (logSheet) {
    logSheet.appendRow([uniqueID, "Pending", ...responses]);
  }
  
  if (studentEmail) {
    MailApp.sendEmail(studentEmail, `Support Ticket Registered: ${uniqueID}`, 
      `Dear Student,\n\nYour support request has been successfully registered with Ticket ID: ${uniqueID}.\n\nOur team will review your concern and update the status shortly. You can use this ID to track your progress on the student portal.\n\nStatus: Pending`);
  }
}

          <h2>Support Tracker</h2>
          <p>Enter your Ticket ID to see the latest update.</p>
          <input type="text" id="tid" placeholder="TKT-XXXX">
          <button onclick="find()">Check Status</button>
          <div id="result"></div>
        </div>
        <script>
          function find() {
            var id = document.getElementById('tid').value;
            var res = document.getElementById('result');
            res.style.display = 'none';
            google.script.run.withSuccessHandler(s => {
              res.innerText = s;
              res.style.display = 'block';
              res.style.background = s.includes('Pending') ? '#fff3cd' : '#d4edda';
            }).lookupTicketStatus(id);
          }
        </script>
      </body>

  `);

function lookupTicketStatus(id) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(LOG_TAB_NAME);
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (data[i][0].toString().trim() === id.trim()) return "Current Status: " + data[i][1];
  }
  return "Ticket Not Found";
}

function lookupEmail(roll) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(DATA_TAB_NAME);
  const data = sheet.getDataRange().getValues();
  for (let i = 2; i < data.length; i++) {
    if (data[i][0].toString().trim() === roll.toString().trim()) return data[i][17]; 
  }
  return null;
}
